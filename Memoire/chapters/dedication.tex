\begin{center}
\textit{À tous les professionnels et chercheurs}\\

\textit{pour leur soutien inconditionnel}\\
\textit{tout au long de cette aventure académique.}
\end{center}

\vspace{2cm}

\begin{center}
\textit{À l'industrie de la cybersécurité,}\\
\textit{qui inspire cette recherche}\\
\textit{par son innovation constante}\\
\textit{et son engagement pour la protection des systèmes critiques.}
\end{center}
